#### 7. DB 연결 관리
```python
# 재시도 로직
max_retries = 2
for attempt in range(max_retries + 1):
    try:
        async with self.connection_pool.acquire() as conn:
            await conn.fetchval('SELECT 1')  # Health check
            result = await conn.fetch(query, *params)
    except Exception as e:
        if "too many clients" in str(e):
            await asyncio.sleep(5)  # 대기 후 재시도
        wait_time = 2 ** attempt  # Exponential backoff
        await asyncio.sleep(wait_time)

# 타임아웃 설정
command_timeout = 600  # 10분
statement_timeout = '600000'  # 10분 (ms)
tcp_keepalives_idle = '30'  # 30초

class USFactorInteractions:
    """
    팩터 간 상호작용 분석으로 숨겨진 시그널 발견

    ⭐ 4가지 Pairwise 상호작용 (한국 모델에 없는 기능!)
    """

    def analyze_interactions(self, factor_scores):
        """
        주요 상호작용 (4개):
        1. Value × Quality (QARP 전략, 가치함정 탐지)
        2. Momentum × Growth (성장 가속, 버블 리스크)
        3. Value × Momentum (역발상 타이밍)
        4. Quality × Growth (지속가능 성장)
        """

    # 1. Value × Quality 상호작용 (35% 가중치)
    def calculate_value_quality(self, value_score, quality_score):
        """
        조합                        | Multiplier | 시그널
        ---------------------------|-----------|--------
        High Value + High Quality   | +25%      | QARP (Quality at Reasonable Price)
        High Value + Low Quality    | -30%      | VALUE_TRAP_RISK
        Low Value + High Quality    | 0~+10%    | QUALITY_PREMIUM
        Low Value + Low Quality     | -20%      | AVOID

        로직:
        if quality >= 0.7:
            multiplier = 1.0 + (0.25 * value * quality)  # Up to +25%
            signal = 'QARP' if value >= 0.6 else 'QUALITY_PREMIUM'
        elif quality < 0.4:
            multiplier = 1.0 - (0.3 * value)  # Up to -30%
            signal = 'VALUE_TRAP_RISK' if value >= 0.6 else 'AVOID'
        """

    # 2. Momentum × Growth 상호작용 (35% 가중치)
    def calculate_momentum_growth(self, momentum_score, growth_score):
        """
        조합                          | Multiplier | 시그널
        -----------------------------|-----------|--------
        High Momentum + High Growth   | +30%      | GROWTH_ACCELERATION
        High Momentum + Low Growth    | -15%      | BUBBLE_RISK
        Low Momentum + High Growth    | +10%      | ACCUMULATION
        Low Momentum + Low Growth     | -25%      | DETERIORATION

        로직:
        if momentum >= 0.7 and growth >= 0.7:
            multiplier = 1.0 + (0.3 * momentum * growth)  # Up to +30%
            signal = 'GROWTH_ACCELERATION'
        elif momentum >= 0.7 and growth < 0.4:
            multiplier = 1.0 - (0.15 * momentum)  # Up to -15%
            signal = 'BUBBLE_RISK'
        """

    # 3. Value × Momentum 상호작용 (15% 가중치)
    def calculate_value_momentum(self, value_score, momentum_score):
        """
        역발상 타이밍 포착

        조합                          | Multiplier | 시그널
        -----------------------------|-----------|--------
        High Value + Rising Momentum  | +20%      | CONTRARIAN_REVERSAL
        High Value + Falling Momentum | -10%      | FALLING_KNIFE
        Low Value + High Momentum     | +10%      | TREND_FOLLOWING

        로직:
        if value >= 0.6 and momentum >= 0.6:
            multiplier = 1.0 + (0.2 * value * momentum)
            signal = 'CONTRARIAN_REVERSAL'
        elif value >= 0.6 and momentum < 0.4:
            multiplier = 1.0 - (0.1 * value)
            signal = 'FALLING_KNIFE'
        """

    # 4. Quality × Growth 상호작용 (15% 가중치)
    def calculate_quality_growth(self, quality_score, growth_score):
        """
        지속가능 성장 평가

        조합                          | Multiplier | 시그널
        -----------------------------|-----------|--------
        High Quality + High Growth    | +25%      | SUSTAINABLE_GROWTH
        High Quality + Low Growth     | +5%       | MATURE_QUALITY
        Low Quality + High Growth     | -20%      | UNSUSTAINABLE

        로직:
        if quality >= 0.7 and growth >= 0.7:
            multiplier = 1.0 + (0.25 * quality * growth)
            signal = 'SUSTAINABLE_GROWTH'
        elif quality < 0.4 and growth >= 0.7:
            multiplier = 1.0 - (0.2 * growth)
            signal = 'UNSUSTAINABLE'
        """

    # 5. Composite Score 계산
    def _calculate_composite(self, interactions):
        """
        전체 팩터 조합 점수 (0-1 범위)

        계산:
        composite = (
            vq_score * 0.35 +      # Value-Quality (35%)
            mg_score * 0.35 +      # Momentum-Growth (35%)
            vm_score * 0.15 +      # Value-Momentum (15%)
            qg_score * 0.15        # Quality-Growth (15%)
        )

        범위: 0-1
        - 1.0: 모든 상호작용 최고 긍정
        - 0.75: 강력 긍정
        - 0.5: 중립
        - 0.25: 강력 부정
        - 0.0: 모든 상호작용 최고 부정
        """

    # 6. Signals 생성
    def _generate_signals(self, interactions):

```

### Step 6: 등급 결정

# 데이터 유효성 검사
if all_factors_zero or confidence_score < 30 or final_score == 0:

### 핵심 혁신: 팩터 상호작용 + 조정 메커니즘

미국 모델의 가장 큰 차별점은 **팩터 상호작용 분석**과 **팩터별 조정 메커니즘**입니다.

```
한국 모델:
각 팩터를 독립적으로 평가하여 단순 가중 합산
→ 팩터 간 시너지/충돌 미반영
→ 팩터 트렌드 변화 미반영

미국 모델:
1. 팩터별 조정 메커니즘 (각 팩터 내부)
   - Quality: Deterioration Detector (-20 ~ +10)
     · 8가지 품질 지표 추세 분석
   - Momentum: Mean Reversion Adjustment (-25 ~ +20)
     · 5가지 지속가능 랠리 조건 체크
   - Growth: Deceleration Detector (-20 ~ +10)
     · 분기별 성장률 추세 + 마진 추세 분석

2. 팩터 상호작용 (4가지 Pairwise)
   - Value × Quality (35% 가중): QARP 전략, 가치함정 탐지
   - Momentum × Growth (35% 가중): 성장 가속, 버블 리스크
   - Value × Momentum (15% 가중): 역발상 타이밍
   - Quality × Growth (15% 가중): 지속가능 성장
   → Composite Score (0-1) → Multiplier (0.5x-1.5x)

3. 최종 점수 증폭/감소
   final_score = base_score × interaction_multiplier

예시 1 (QARP 발견):
- Deep Value (30점, adj) + High Quality (85점, adj)
  → 상호작용: QARP (+25% boost)
  → 한국: (30+85)/2 = 57.5점
  → 미국: 57.5점 × 1.4 = 80.5점 (바겐세일!)

예시 2 (가치함정 탐지):
- High Value (85점, adj) + Low Quality (30점, deteriorating -15)
  → Quality 조정: 30-15 = 15점
  → 상호작용: VALUE_TRAP_RISK (-30% penalty)
  → 한국: (85+30)/2 = 57.5점
  → 미국: (85+15)/2 × 0.7 = 35점 (함정!)

예시 3 (버블 리스크):
- High Momentum (85점, overbought -10) + Low Growth (25점, decelerating -15)
  → 조정 후: 75점 + 10점 = 42.5점
  → 상호작용: BUBBLE_RISK (-15% penalty)
  → 한국: (85+25)/2 = 55점
  → 미국: 42.5점 × 0.85 = 36점 (위험!)
```
